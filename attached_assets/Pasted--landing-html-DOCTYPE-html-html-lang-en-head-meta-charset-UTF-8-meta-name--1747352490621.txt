<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Install Dear Teddy</title>
  <link href="https://cdn.jsdelivr.net/npm/tailwindcss@2.2.19/dist/tailwind.min.css" rel="stylesheet">
</head>
<body class="bg-white text-gray-800 flex flex-col justify-center items-center min-h-screen px-4">
  <h1 class="text-2xl font-semibold mb-4">Install Dear Teddy</h1>
  <p class="mb-6 text-center max-w-md">This app lives on your home screen and helps you reflect anytime. Tap below to begin installation.</p>
  <button id="installBtn" class="bg-blue-600 text-white px-5 py-3 rounded-lg shadow hover:bg-blue-700" style="display: none;">Install Dear Teddy</button>
  <p id="fallback" class="mt-4 text-sm text-gray-600 hidden">If nothing happens, your browser might not support app installation.</p>

  <script>
    let deferredPrompt;
    const installBtn = document.getElementById('installBtn');
    const fallback = document.getElementById('fallback');

    window.addEventListener('beforeinstallprompt', (e) => {
      e.preventDefault();
      deferredPrompt = e;
      installBtn.style.display = 'block';

      installBtn.addEventListener('click', () => {
        deferredPrompt.prompt();
        installBtn.style.display = 'none';
      });
    });

    window.addEventListener('load', () => {
      setTimeout(() => {
        if (!installBtn.style.display || installBtn.style.display === 'none') {
          fallback.classList.remove('hidden');
        }
      }, 3000);
    });
  </script>
</body>
</html>
